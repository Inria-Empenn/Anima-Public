project(${PROJECT_PREFIX}NyulStandardization)

## List Sources
## #############################################################################

)

## add executable
## #############################################################################

  ${${PROJECT_NAME}_CFILES}
)

## #############################################################################

target_link_libraries(${PROJECT_NAME}
  ${ITKIO_LIBRARIES}
  ITKStatistics
  AnimaCore
)

## #############################################################################
## install
## #############################################################################

set_exe_install_rules(${PROJECT_NAME})
